# Emotion

---

There are various ways Cijul can be used alongside Emotion.

## CSS Prop

<!-- <CSSProp>Hello</CSSProp> -->

```jsx
/** @jsx jsx */
import { jsx, css } from '@emotion/core'
import { color, text } from '@nearform/cijul'

export const Header = ({ children }) => (
  <h1
    css={css`
      ${color.red}
      ${text.size.large}
    `}
  >
    {children}
  </h1>
)
```

## Array

<!-- <Array>Hello</Array> -->

```jsx
/** @jsx jsx */
import { jsx } from '@emotion/core'
import { color, text } from '@nearform/cijul'

export const Header = ({ children }) => (
  <h1 css={[color.red, text.size.large]}>{children}</h1>
)
```

## Object Literal

<!-- <ObjectLiteral>Hello</ObjectLiteral> -->

```jsx
/** @jsx jsx */
import { jsx } from '@emotion/core'
import { color, text } from '@nearform/cijul'

export const Header = ({ children }) => (
  <h1
    css={{
      ...color.red,
      ...text.size.large
    }}
  >
    {children}
  </h1>
)
```

## Styled

<!-- <Styled>Hello</Styled> -->

```jsx
import styled from '@emotion/styled'
import { color, text } from '@nearform/cijul'

const StyledHeader = styled.h1`
  ${color.red}
  ${text.size.large}
`

export const Header = ({ children }) => <StyledHeader>{children}</StyledHeader>
```

## Styled Object Literal

<!-- <StyledObjectLiteral>Hello</StyledObjectLiteral> -->

```jsx
import styled from '@emotion/styled'
import { color, text } from '@nearform/cijul'

const StyledHeader = styled.h1({
  ...color.red,
  ...text.size.large
})

export const Header = ({ children }) => <StyledHeader>{children}</StyledHeader>
```
