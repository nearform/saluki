import { Wrap, Box } from './'

# Flex Wrap

---

## Example

<Wrap>
  <Box light />
  <Box />
  <Box />
</Wrap>

```jsx
const Container = styled.div`
  ${layout.wrap.wrap}
  ${layout.display.flex}
`

const Box = styled.div`
  ${size.height.medium}
  ${size.width.half}
  ${({ light }) => (light ? background.red : background.black)};
`
```

## Rules

```js
layout.wrap.noWrap
layout.wrap.wrap
layout.wrap.reverse
```

## Defaults

```js
{
  layout: {
    wrap: {
      noWrap: 'nowrap',
      wrap: 'wrap',
      reverse: 'wrap-reverse'
    }
  }
}
```
