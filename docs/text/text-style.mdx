import { Style } from './'

# Text Style

---

## Example

<Style>Hello</Style>

```jsx
const Text = styled.h1`
  ${text.style.italic}
`
```

## Rules

```js
text.style.italic
text.style.oblique
text.style.normal
```

## Defaults

```js
{
  text: {
    style: {
      normal: 'normal',
      italic: 'italic',
      oblique: 'oblique'
    }
  }
}
```
